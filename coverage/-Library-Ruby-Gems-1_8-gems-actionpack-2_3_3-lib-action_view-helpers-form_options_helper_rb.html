231       # NOTE: Only the option tags are returned, you have to wrap this call in a regular HTML select tag.
 
232       def options_for_select(container, selected = nil)
 
233         return container if String === container
 
234 
235         container = container.to_a if Hash === container
 
236         selected, disabled = extract_selected_and_disabled(selected)
 
237 
238         options_for_select = container.inject([]) do |options, element|
 
239           text, value = option_text_and_value(element)
 
240           selected_attribute = ' selected="selected"' if option_value_selected?(value, selected)
 
241           disabled_attribute = ' disabled="disabled"' if disabled && option_value_selected?(value, disabled)
 
242           options << %(<option value="#{html_escape(value.to_s)}"#{selected_attribute}#{disabled_attribute}>#{html_escape(text.to_s)}</option>)
 
243         end
 
244 
245         options_for_select.join("\n")
 
246       end

248       # Returns a string of option tags that have been compiled by iterating over the +collection+ and assigning the
 
249       # the result of a call to the +value_method+ as the option value and the +text_method+ as the option text.
 
250       # Example:
 
251       #   options_from_collection_for_select(@people, 'id', 'name')
 
252       # This will output the same HTML as if you did this:
 
253       #   <option value="#{person.id}">#{person.name}</option>
 
254       #
 
255       # This is more often than not used inside a #select_tag like this example:
 
256       #   select_tag 'person', options_from_collection_for_select(@people, 'id', 'name')
 
257       #
 
258       # If +selected+ is specified as a value or array of values, the element(s) returning a match on +value_method+
 
259       # will be selected option tag(s).
 
260       #
 
261       # If +selected+ is specified as a Proc, those members of the collection that return true for the anonymous
 
262       # function are the selected values.
 
263       #
 
264       # +selected+ can also be a hash, specifying both <tt>:selected</tt> and/or <tt>:disabled</tt> values as required.
 
265       #
 
266       # Be sure to specify the same class as the +value_method+ when specifying selected or disabled options.
 
267       # Failure to do this will produce undesired results. Example:
 
268       #   options_from_collection_for_select(@people, 'id', 'name', '1')
 
269       # Will not select a person with the id of 1 because 1 (an Integer) is not the same as '1' (a string)
 
270       #   options_from_collection_for_select(@people, 'id', 'name', 1)
 
271       # should produce the desired results.
 
272       def options_from_collection_for_select(collection, value_method, text_method, selected = nil)
 
273         options = collection.map do |element|
 
274           [element.send(text_method), element.send(value_method)]
 
275         end
 
276         selected, disabled = extract_selected_and_disabled(selected)
 
277         select_deselect = {}
 
278         select_deselect[:selected] = extract_values_from_collection(collection, value_method, selected)
 
279         select_deselect[:disabled] = extract_values_from_collection(collection, value_method, disabled)
 
280 
281         options_for_select(options, select_deselect)
 
282       end

395       # Returns a string of option tags for pretty much any time zone in the
 
396       # world. Supply a TimeZone name as +selected+ to have it marked as the
 
397       # selected option tag. You can also supply an array of TimeZone objects
 
398       # as +priority_zones+, so that they will be listed above the rest of the
 
399       # (long) list. (You can use TimeZone.us_zones as a convenience for

401       # of your choice)

403       # The +selected+ parameter must be either +nil+, or a string that names
 
404       # a TimeZone.
 
405       #
 
406       # By default, +model+ is the TimeZone constant (which can be obtained
 
407       # in Active Record as a value object). The only requirement is that the
 
408       # +model+ parameter be an object that responds to +all+, and returns
 
409       # an array of objects that represent time zones.

411       # NOTE: Only the option tags are returned, you have to wrap this call in
 
412       # a regular HTML select tag.
 
413       def time_zone_options_for_select(selected = nil, priority_zones = nil, model = ::ActiveSupport::TimeZone)
 
414         zone_options = ""
 
415 
416         zones = model.all
 
417         convert_zones = lambda { |list| list.map { |z| [ z.to_s, z.name ] } }
 
418 
419         if priority_zones
 
420 	        if priority_zones.is_a?(Regexp)
 
421             priority_zones = model.all.find_all {|z| z =~ priority_zones}
 
422 	        end
 
423           zone_options += options_for_select(convert_zones[priority_zones], selected)
 
424           zone_options += "<option value=\"\" disabled=\"disabled\">-------------</option>\n"
 
425 
426           zones = zones.reject { |z| priority_zones.include?( z ) }
 
427         end
 
428 
429         zone_options += options_for_select(convert_zones[zones], selected)
 
430         zone_options
 
431       end
